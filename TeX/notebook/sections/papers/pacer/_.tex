\newcommand{\State}[2]{%
    % #1 : x
    % #2 : y
    \def\p{0.02} % Padding
    \begin{scope}
        \filldraw[fill=Blue!25]
        ($(#1,#2)+(-1+\p,0.5-\p)$)
        rectangle
        ($(#1,#2)+(1-\p,-0.5+\p)$);
        \node at (#1,#2) {State};
    \end{scope}
}

\newcommand{\Action}[2]{%
    % #1 : x
    % #2 : y
    \begin{scope}
        \filldraw[fill=Green!25]
        ($(#1,#2)+(-1+\p,0.5-\p)$)
        rectangle
        ($(#1,#2)+(1-\p,-0.5+\p)$);
        \node at (#1,#2) {Action};
    \end{scope}
}

    \def\xleft{0}
