\section{Introduction}

\begin{frame}{Motivation}

    \textbf{Imitation Learning (IL)}
    \begin{itemize}
        \item Learn robot policies directly from demonstrations
        \item Avoids hand-crafted rewards
        \item Effective for manipulation and trajectory tasks
    \end{itemize}

    \vspace{0.5em}

    \textbf{Practical Challenges}
    \begin{itemize}
        \item \textbf{Pace variability}
            \begin{itemize}
                \item Same task executed at different speeds
            \end{itemize}

        \item \textbf{Local corruptions}
            \begin{itemize}
                \item Brief unintended deviations
                \item Fatigue, hesitation, over-corrections
            \end{itemize}
    \end{itemize}

    \vspace{0.5em}

    \textbf{Problem:}
    Standard Behavioral Cloning treats all samples equally.

\end{frame}

\begin{frame}{Goals of PACER}

    \begin{block}{Objective}
        Learn robust policies from sparse and imperfect demonstrations.
    \end{block}

    \textbf{PACER Contributions}
    \begin{enumerate}
        \item Learn a latent \textbf{task phase} for alignment
        \item Detect corrupted segments using:
            \begin{itemize}
                \item Median
                \item MAD
                \item Tukey Biweight
            \end{itemize}
        \item Construct refined pseudo-labels before policy training
    \end{enumerate}

    \vspace{0.5em}

    \begin{alertblock}{Key Idea}
        Move all robustness into a preprocessing stage while keeping policy training simple.
    \end{alertblock}

\end{frame}

\begin{frame}{Phase Alignment}

    \textbf{Problem:}
    Different demonstrations progress at different speeds.

    \vspace{0.7em}

    \textbf{Solution:}
    Learn a state-dependent phase variable
    \[
        \tau \in [0,1]
    \]

    \vspace{0.7em}

    \textbf{Ranking Objective}
    \[
        \mathcal{L}_{\mathrm{rank}}
        =
        \sum_{t>t'}
        \log\left(
            1+\exp\left(
                m-(g(x_t)-g(x_{t'}))
            \right)
        \right)
    \]

    \vspace{0.7em}

    \begin{itemize}
        \item Later states should have higher phase
        \item Independent of absolute time
        \item Enables progress-based synchronization
    \end{itemize}

\end{frame}

\begin{frame}{Robust Consensus Statistics}

    \textbf{Phase Binning}
    \begin{itemize}
        \item Divide phase interval into bins
        \item Samples with similar progress are grouped together
    \end{itemize}

    \textbf{Per-bin Robust Statistics}
    \[
        \alpha_a[b] = \mathrm{median}(a_{i,t})
    \]
    \[
        \beta_a[b] = \mathrm{median}(\|a_{i,t}\|)
    \]

    \textbf{Why Median?}
    \begin{itemize}
        \item Resistant to outliers
        \item Captures typical behavior
    \end{itemize}

    \textbf{Tangents}
    \begin{itemize}
        \item Finite differences across bins
        \item Capture canonical direction of motion
    \end{itemize}

\end{frame}

\begin{frame}{Trust Estimation}

    \textbf{Leave-One-Out Consensus}
    \begin{itemize}
        \item A demo cannot validate itself
    \end{itemize}

    \vspace{0.7em}

    \textbf{Residual}
    \[
        r_{i,t} = \|a_{i,t} - \alpha_a^{(-i)}[b]\|
    \]

    \vspace{0.7em}

    \textbf{Robust z-score}
    \[
        z_{i,t}
        =
        \frac{r_{i,t}}
        {\mathrm{MAD}[b]+\varepsilon}
    \]

    \vspace{0.7em}

    \textbf{Tukey Biweight}
    \[
        w_{i,t}
        =
        {\left(1-{(z/c)}^2\right)}^2
    \]

    \vspace{0.7em}

    \begin{itemize}
        \item High agreement $\rightarrow$ high trust
        \item Outliers $\rightarrow$ low trust
    \end{itemize}

\end{frame}

\begin{frame}{Pseudo-Label Refinement}

    For each sample:

    \vspace{0.5em}

    \begin{enumerate}
        \item \textbf{Debias toward median anchor}
            \[
                y^{(1)}
                =
                \gamma a
                +(1-\gamma)\alpha_a
            \]

            \vspace{0.5em}

        \item \textbf{Directional alignment}
            \begin{itemize}
                \item Align with ribbon tangent
            \end{itemize}

            \vspace{0.5em}

        \item \textbf{Sideways attenuation}
            \begin{itemize}
                \item Suppress lateral deviations
            \end{itemize}

            \vspace{0.5em}

        \item \textbf{Speed regularization}

            \vspace{0.5em}

        \item \textbf{Temporal smoothing}
    \end{enumerate}

\end{frame}

\begin{frame}{Policy Training}

    \textbf{Final Objective}
    \[
        \mathcal{L}
        =
        \mathbb{E}_{i,t}
        \left[
            w_{i,t}
            \cdot
            \mathrm{Huber}
            \left(
                f_\theta(x_{i,t}) - y^*_{i,t}
            \right)
        \right]
    \]

    \vspace{1em}

    \textbf{Key Point}
    \begin{itemize}
        \item Training remains standard behavioral cloning
        \item Robustness handled during preprocessing
    \end{itemize}

    \vspace{1em}

    \begin{block}{Benefits}
        \begin{itemize}
            \item Stable training
            \item Noise resilience
            \item Interpretable diagnostics
        \end{itemize}
    \end{block}

\end{frame}

\begin{frame}
    \frametitle{\textbf{Limitations}}

    \begin{block}
        {\textbf{Limitations}}

        \begin{itemize}
            \item Requires meaningful phase estimation
            \item Works best for local corruptions
            \item Sensitive to hyperparameters
            \item Sparse bins reduce robustness
        \end{itemize}
    \end{block}

    \begin{block}
        {\textbf{Future Work}}

        \begin{itemize}
            \item Uncertainty-aware consensus models
            \item Vision-based phase estimation
            \item Diffusion policies
            \item Real robot deployment
        \end{itemize}
    \end{block}

    \end{block}
