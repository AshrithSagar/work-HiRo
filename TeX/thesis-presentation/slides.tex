\section{Introduction}

% -----------------------------------------------------
\begin{frame}{Introduction}{Imitation Learning \& Behavioral Cloning}

    \begin{columns}[T,onlytextwidth]

        % ---------------- LEFT COLUMN ----------------
        \column{0.45\textwidth}

        \textbf{Goal}
        \begin{itemize}
            \item Learning from demonstrations (LfD)
            \item Avoid reward engineering
        \end{itemize}

        \vspace{0.5em}

        \textbf{Applications}
        \begin{itemize}
            \item Robotics
            \item Autonomous driving
            \item Human-robot interaction
        \end{itemize}

        \vspace{0.45em}

        \begin{block}{Behavioral Cloning}
            Imitates expert state-action pairs.
            \[
                f_\theta: x_t \rightarrow a_t
            \]
            \[
                \pi(a \mid s) \approx \pi_E(a \mid s)
            \]
        \end{block}

        % ---------------- RIGHT COLUMN ----------------
        \column{0.5\textwidth}

        \begin{alertblock}{Key Limitation}
            Assumes all expert actions are correct
        \end{alertblock}

        \vspace{0.5em}

        \textbf{Reality of Data}
        \begin{itemize}
            \item Fatigue / hesitation
            \item Distractions
            \item Over-corrections
        \end{itemize}

        \vspace{0.5em}

        \textbf{Consequences}
        \begin{itemize}
            \item Learns noisy actions
            \item Error compounding in policy
        \end{itemize}

    \end{columns}

\end{frame}

\begin{frame}{Challenges}{2. Local Corruptions}

    \textbf{Problem}
    \begin{itemize}
        \item \textit{Short segments} may contain unintended deviations
        \item Operator bias / fatigue / hesitations / distractions / over-corrections
        \item Corruptions are \textbf{local}, not global
    \end{itemize}

    \vspace{1em}

    \begin{itemize}
        \item Becomes especially crucial, when data is sparse.
    \end{itemize}

\begin{frame}
    \frametitle{\textbf{PACER}}
    \framesubtitle{\textbf{P}rogress \textbf{A}ligned \textbf{C}uration for \textbf{E}rror \textbf{R}esilient Imitation Learning}

    \begin{block}{Objective}
        Learn robust policies from sparse and imperfect demonstrations.
    \end{block}

    \vspace{0.75em}

    \begin{enumerate}
        \item Phase alignment: Learn a latent task phase for temporal alignment
        \item Robust statistics: Detect corrupted segments
        \item Repair corrupted labels before policy training
    \end{enumerate}

    \vspace{0.75em}

    \begin{alertblock}{Key Design Principle}
        Move all robustness into \textbf{preprocessing} while keeping policy training \textit{simple}.
    \end{alertblock}
\end{frame}

\begin{frame}
    \frametitle{\textbf{Phase Alignment}}

    \textbf{Idea:}
    Learn a latent task phase $\tau$ using a scoring function $g_\psi$:
    \[
        \tau \leftarrow g_\psi(x)
    \]
    Represents task progress independent of time.
    Depends only on state.

    \begin{itemize}
        \item Align demonstrations running at different speeds
        \item Compare equivalent task stages
        \item Build phase-local consensus
    \end{itemize}

    We use a simple MLP with one hidden layer to score, and enfore monotonicity
    \[
        g_\psi(x_{i, t}) > g_\psi(x_{i, t'}), \quad \forall t > t'
    \]

    \begin{block}{Monotone Ranking Loss}
        \[
            \mathcal{L}_{\mathrm{rank}}
            =
            \sum_{i}
            \sum_{t>t'}
            \log
            \left(
                1+
                \exp{\left(m-(g_\psi(x_{i, t})-g_\psi(x_{i, t'}))\right)}
            \right)
        \]
        \small
        Later states should receive larger phase scores.
    \end{block}

\end{frame}

\begin{frame}{Phase Bins}

    \textbf{Binning}
    \begin{itemize}
        \item Divide the phase interval $[0, 1]$ into $B$ equally-spaced bins.
            \begin{itemize}
                \item Samples with similar progress are grouped together
                \item Enables direct comparison across demonstrations
                \item Removes dependence on time index
            \end{itemize}
            \vspace{-0.5em}
            \[
                \mathcal{I}_b = \left\{(i, t) \Big| \tau_{i, t} \in \left[{\frac{b}{B}}, {\frac{b+1}{B}}\right) \right\}
            \]
    \end{itemize}

    \textbf{Robust consensus statistics}
    \begin{itemize}
        \item \small Within each bin, compute consensus statistics.
            \vspace{-0.75em}
            \begin{align}
                \alpha_a[b] & = \mathrm{median}\{a_{i,t} : (i,t)\in\mathcal{I}_b\}, \\
                \alpha_s[b] & = \mathrm{median}\{x_{i,t} : (i,t)\in\mathcal{I}_b\}, \\
                \beta_a[b]  & = \mathrm{median}\{\|a_{i,t}\| : (i,t)\in\mathcal{I}_b\}, \\
                \beta_s[b]  & = \mathrm{median}\{\|\dot{x}_{i,t}\| : (i,t)\in\mathcal{I}_b\}.
            \end{align}
            \vspace{-1.5em}
        \item \small Finite differencing per-bin medians $\rightarrow$ Action, State tangents $t_a[b]$, $t_s[b]$
        \item \small Capturing the canonical direction of motion at each task phase.
    \end{itemize}

\end{frame}

\begin{frame}
    \frametitle{\textbf{Robust Trust Estimation}}

    \textbf{Leave-One-Out Residual}
    \[
        r_{i,t}
        =
        \|a_{i,t} - \alpha_a^{(-i)}[b]\|
    \]

    \small
    Leave-one-out consensus prevents demonstrations from validating their own corruptions.

    \vspace{0.75em}

    \textbf{Robust z-score}
    \[
        z_{i,t}
        =
        \frac{
            r_{i,t}
        }{
            \mathrm{MAD}[b] + \varepsilon
        }
    \]

    \vspace{0.75em}

    \textbf{Tukey Trust Weight}
    \[
        w_{i,t}
        =
        \left(
            1-(z/c)^2
        \right)^2
    \]

    \vspace{1em}

    \begin{columns}

        \column{0.48\linewidth}

        \begin{block}{High Trust}
            Small deviation from phase consensus
        \end{block}

        \column{0.48\linewidth}

        \begin{alertblock}{Low Trust}
            Corrupted or inconsistent segment
        \end{alertblock}

    \end{columns}

\end{frame}

\begin{frame}
    \frametitle{\textbf{Pseudo-label Refinement}}

    \centering
    \vspace{-2em}
    \[
        \begin{aligned}
            a_{i,t}
            &
            \rightarrow
            \text{debias}
            \rightarrow
            \text{align}
            \rightarrow
            \text{smooth}
            \rightarrow
            y^*_{i,t}
            \\
            x_{i,t}
            &
            \rightarrow
            \text{debias}
            \rightarrow
            \text{align}
            \rightarrow
            \text{smooth}
            \rightarrow
            x^*_{i,t}
        \end{aligned}
    \]

    \vspace{-0.5em}
    \begin{enumerate}
        \item Debias toward median anchor
        \item Directional alignment
        \item Sideways attenuation
        \item Speed regularization
        \item Temporal smoothing
    \end{enumerate}

    \begin{block}{Consensus Debiasing}

        \[
            y^*
            =
            \gamma a
            +
            (1-\gamma)\alpha_a[b]
        \]

    \end{block}

    \vspace{1em}

    \begin{itemize}
        \item Pull low-trust actions toward phase consensus
        \item Preserve dominant motion direction
        \item Smooth uncertain trajectory regions
    \end{itemize}

\end{frame}

% =====================================================
% SLIDE 6 : FINAL OBJECTIVE
% =====================================================

\begin{frame}{Final Training Objective}

    Train a simple weighted BC policy using the computed trusts and refined pseudo-labels.
    \[
        \mathcal{L}
        =
        \mathbb{E}_{i,t}
        \left[
            w_{i,t}
            \cdot
            \mathrm{Huber}
            (
                f_\theta(x_{i,t})
                -
                y^*_{i,t}
            )
        \right]
    \]

    \vspace{1.5em}

    PACER modifies the supervision targets, not the policy architecture.

\end{frame}

\begin{frame}{Discussion}

    \textbf{Why PACER Works}

    \begin{enumerate}
        \item Progress alignment compares equivalent task stages
        \item Robust statistics resist outliers
        \item Leave-one-out prevents self-validation
        \item Pseudo-label repair suppresses local corruptions
    \end{enumerate}

    \vspace{1em}

    \begin{block}{Interpretability}
        PACER also provides useful diagnostics:
        \begin{itemize}
            \item Trust maps
            \item MAD uncertainty regions
            \item Phase-aligned consensus trajectories
        \end{itemize}
    \end{block}

\end{frame}

\section{Limitations and Future Work}

% -----------------------------------------------------
\begin{frame}{Limitations}

    \begin{itemize}
        \item Requires meaningful phase estimation
            \begin{itemize}
                \item Benefits most when deviations are \textit{short}, relative to the trajectory
                \item Breaks down under \textit{long-horizon tasks} or \textit{globally biased corruptions} or when progress is poorly estimated
            \end{itemize}
        \item Assumes mostly local corruptions
        \item Sensitive to hyperparameters; Bias-Variance trade-offs
        \item Sparse bins reduce robustness
            \begin{itemize}
                \item Very few demonstrations weaken LOO consensus
            \end{itemize}

    \end{itemize}

\end{frame}

% -----------------------------------------------------
\begin{frame}{Future Work}

    \begin{itemize}
        \item Vision-based phase estimation
        \item Uncertainty-aware consensus models
        \item Integration with diffusion policies
        \item Online/Iterative refinement with environment interaction
        \item Validation on real robot systems
    \end{itemize}

\end{frame}

\begin{frame}{Median Absolute Deviation (MAD)}

    \textbf{Need}

    Estimate spread robustly.

    \vspace{1em}

    \[
        \mathrm{MAD}
        =
        1.4826
        \cdot
        \mathrm{median}
        |v_i - \mathrm{median}(v)|
    \]

    \vspace{1em}

    \textbf{Advantages}

    \begin{itemize}
        \item Robust alternative to variance
        \item Stable under outliers
        \item Enables robust z-scores
    \end{itemize}

\end{frame}
