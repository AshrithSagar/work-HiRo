\section{Introduction}

% -----------------------------------------------------
\begin{frame}{Imitation Learning}

    \textbf{Goal}
    \begin{itemize}
        \item Learn robot behavior directly from demonstrations
        \item Avoid manual reward engineering
        \item Reduce task-specific controller design
    \end{itemize}

    \vspace{0.5em}

    \textbf{Applications}
    \begin{itemize}
        \item Robotic manipulation
        \item Autonomous driving
        \item Surgical robotics
        \item Human-robot collaboration
    \end{itemize}

    \vspace{0.75em}

    \begin{block}{Behavioral Cloning}
        Learn a policy mapping:
        \[
            f_\theta : x_t \rightarrow a_t
        \]
        using demonstrated state-action pairs.
    \end{block}

\end{frame}

% -----------------------------------------------------
\begin{frame}{Why Standard Behavioral Cloning Fails}

    \textbf{Behavioral Cloning Assumption}
    \begin{itemize}
        \item Every demonstrated action is correct
        \item All samples contribute equally
    \end{itemize}

    \vspace{0.75em}

    \begin{alertblock}{Reality}
        Human demonstrations are imperfect.
    \end{alertblock}

    \vspace{0.5em}

    \begin{itemize}
        \item Operator fatigue
        \item Hesitation
        \item Momentary distractions
        \item Over-corrections
    \end{itemize}

    \vspace{0.75em}

    \textbf{Result}
    \begin{itemize}
        \item BC learns corrupted actions directly
        \item Errors propagate into the learned policy
    \end{itemize}

\end{frame}

% -----------------------------------------------------
\begin{frame}{Challenge 1: Pace Variability}

    \textbf{Problem}
    \begin{itemize}
        \item Different demonstrations execute at different speeds
        \item Same task phase appears at different timestamps
    \end{itemize}

    \vspace{1em}

    \centering

    \begin{tikzpicture}[>=Stealth]

        \draw[thick,->] (0,0) -- (10,0) node[right]{Time};

        \draw[blue, thick] (0,1) -- (2,1) -- (5,1) -- (8,1);
        \node[left] at (0,1) {Demo A};

        \draw[red, thick] (0,2) -- (4,2) -- (7,2) -- (9,2);
        \node[left] at (0,2) {Demo B};

        \draw[dashed] (4,0.5) -- (4,2.5);

        \node at (2,2.7) {Same semantic stage};

    \end{tikzpicture}

    \vspace{1em}

    \begin{alertblock}{Key Observation}
        Time index is not a reliable measure of task progress.
    \end{alertblock}

\end{frame}

% -----------------------------------------------------
\begin{frame}{Challenge 2: Local Corruptions}

    \textbf{Problem}
    \begin{itemize}
        \item Short segments may contain unintended deviations
        \item Corruptions are local, not global
    \end{itemize}

    \vspace{0.75em}

    \begin{center}
        \includegraphics[width=0.8\linewidth]{../thesis-report/figs/b1.jpeg}
    \end{center}

    \vspace{0.5em}

    \begin{itemize}
        \item Standard BC treats corrupted and clean samples equally
        \item Even brief corruptions can distort the learned policy
    \end{itemize}

\end{frame}

% -----------------------------------------------------
\begin{frame}{Goals of PACER}

    \begin{block}{Objective}
        Learn robust policies from sparse and imperfect demonstrations.
    \end{block}

    \vspace{0.75em}

    \textbf{PACER Contributions}
    \begin{enumerate}
        \item Learn a latent task phase for temporal alignment
        \item Detect corrupted segments using robust statistics
        \item Repair corrupted labels before policy training
    \end{enumerate}

    \vspace{0.75em}

    \begin{alertblock}{Key Design Principle}
        Move all robustness into preprocessing while keeping policy training simple.
    \end{alertblock}

\end{frame}

% -----------------------------------------------------
\begin{frame}{Problem Formulation}

    Given demonstrations:

    \[
        D_i = \{(x_{i,t}, a_{i,t})\}_{t=1}^{T_i}
    \]

    where:

    \begin{itemize}
        \item $x_{i,t}$ : robot state
        \item $a_{i,t}$ : demonstrated action
        \item demonstrations may contain corruptions
    \end{itemize}

    \vspace{0.75em}

    \textbf{Goal}

    \begin{block}{Learn}
        A policy:
        \[
            f_\theta(x_t) \rightarrow a_t
        \]
        that reproduces intended behavior while ignoring corrupted segments.
    \end{block}

\end{frame}

% -----------------------------------------------------
\begin{frame}{Phase Alignment}

    \textbf{Solution}

    Learn a latent phase variable:

    \[
        \tau \in [0,1]
    \]

    representing task progress independently of time.

    \vspace{1em}

    \textbf{Benefits}
    \begin{itemize}
        \item Align demonstrations executed at different speeds
        \item Compare samples at equivalent task stages
        \item Build progress-based consensus
    \end{itemize}

\end{frame}

% -----------------------------------------------------
\begin{frame}{Pairwise Ranking Loss}

    \textbf{Monotonicity Constraint}

    Later states should receive larger phase scores.

    \vspace{1em}

    \[
        \mathcal{L}_{\mathrm{rank}}
        =
        \sum_{t>t'}
        \log\left(
            1+
            \exp\left(
                m-(g(x_t)-g(x_{t'}))
            \right)
        \right)
    \]

    \vspace{1em}

    \begin{itemize}
        \item Enforces ordered progression
        \item Independent of execution speed
        \item Produces a smooth phase coordinate
    \end{itemize}

\end{frame}

% -----------------------------------------------------
\begin{frame}{Estimated Task Phases}

    \centering

    \includegraphics[width=0.95\linewidth]{../thesis-report/figs/a2.jpeg}

    \vspace{0.5em}

    \small
    Learned phases remain approximately monotonic despite varying execution speeds.

\end{frame}

% -----------------------------------------------------
\begin{frame}{Phase Binning}

    \textbf{Idea}

    Divide the phase interval into bins.

    \[
        \tau \in [0,1]
    \]

    \vspace{0.75em}

    \begin{itemize}
        \item Samples with similar progress are grouped together
        \item Enables direct comparison across demonstrations
        \item Removes dependence on time index
    \end{itemize}

    \vspace{1em}

    \begin{block}{Within each bin}
        Compute robust consensus statistics.
    \end{block}

\end{frame}

% -----------------------------------------------------
\begin{frame}{Robust Consensus Statistics}

    \textbf{Per-bin Statistics}

    \[
        \alpha_a[b] = \mathrm{median}(a_{i,t})
    \]

    \[
        \beta_a[b] = \mathrm{median}(\|a_{i,t}\|)
    \]

    \vspace{1em}

    \textbf{Why Median?}

    \begin{itemize}
        \item Robust to outliers
        \item Resistant to corrupted segments
        \item Captures typical behavior
    \end{itemize}

    \vspace{0.75em}

    \begin{alertblock}{Observation}
        Means shift heavily under corruptions, medians remain stable.
    \end{alertblock}

\end{frame}

% -----------------------------------------------------
\begin{frame}{Median Absolute Deviation (MAD)}

    \textbf{Need}

    Estimate spread robustly.

    \vspace{1em}

    \[
        \mathrm{MAD}
        =
        1.4826
        \cdot
        \mathrm{median}
        |v_i - \mathrm{median}(v)|
    \]

    \vspace{1em}

    \textbf{Advantages}

    \begin{itemize}
        \item Robust alternative to variance
        \item Stable under outliers
        \item Enables robust z-scores
    \end{itemize}

\end{frame}

% -----------------------------------------------------
\begin{frame}{Leave-One-Out Consensus}

    \textbf{Problem}

    A corrupted trajectory should not validate itself.

    \vspace{1em}

    \begin{columns}

        \column{0.48\linewidth}

        \textbf{Without LOO}
        \begin{itemize}
            \item Corrupted sample contributes to consensus
            \item Outlier partially hides itself
        \end{itemize}

        \column{0.48\linewidth}

        \textbf{With LOO}
        \begin{itemize}
            \item Remove current demonstration
            \item Compare only against others
        \end{itemize}

    \end{columns}

    \vspace{1em}

    \begin{block}{Result}
        More reliable corruption detection.
    \end{block}

\end{frame}

% -----------------------------------------------------
\begin{frame}{Trust Estimation}

    \textbf{Residual}

    \[
        r_{i,t}
        =
        \|a_{i,t} - \alpha_a^{(-i)}[b]\|
    \]

    \vspace{1em}

    \textbf{Robust z-score}

    \[
        z_{i,t}
        =
        \frac{r_{i,t}}{\mathrm{MAD}[b]+\varepsilon}
    \]

    \vspace{1em}

    \textbf{Interpretation}

    \begin{itemize}
        \item Small deviation $\rightarrow$ high trust
        \item Large deviation $\rightarrow$ low trust
    \end{itemize}

\end{frame}

% -----------------------------------------------------
\begin{frame}{Tukey Biweight}

    \[
        w_{i,t}
        =
        \left(1-(z/c)^2\right)^2
    \]

    \vspace{1em}

    \textbf{Behavior}

    \begin{itemize}
        \item Near consensus $\rightarrow$ weight close to 1
        \item Moderate deviation $\rightarrow$ smoothly reduced weight
        \item Large outlier $\rightarrow$ weight close to 0
    \end{itemize}

    \vspace{1em}

    \begin{alertblock}{Why Tukey?}
        Completely suppresses gross outliers while preserving smoothness.
    \end{alertblock}

\end{frame}

% -----------------------------------------------------
\begin{frame}{Pseudo-Label Refinement}

    \textbf{Goal}

    Repair corrupted labels before training.

    \vspace{1em}

    \begin{enumerate}
        \item Debias toward median anchor
        \item Directional alignment
        \item Sideways attenuation
        \item Speed regularization
        \item Temporal smoothing
    \end{enumerate}

\end{frame}

% -----------------------------------------------------
\begin{frame}{Debiasing and Directional Alignment}

    \textbf{Debias Toward Consensus}

    \[
        y^{(1)}
        =
        \gamma a
        +(1-\gamma)\alpha_a
    \]

    \vspace{1em}

    \textbf{Directional Alignment}

    \begin{itemize}
        \item Compute canonical ribbon tangent
        \item Align actions with dominant motion direction
    \end{itemize}

    \vspace{0.75em}

    \begin{block}{Effect}
        Suppresses spurious directional deviations.
    \end{block}

\end{frame}

% -----------------------------------------------------
\begin{frame}{Speed Regularization and Smoothing}

    \textbf{Speed Regularization}

    \begin{itemize}
        \item Blend action magnitude toward consensus speed
        \item Remove anomalous spikes
    \end{itemize}

    \vspace{1em}

    \textbf{Temporal Smoothing}

    \begin{itemize}
        \item Apply exponential moving average
        \item Reduce abrupt transitions
        \item Improve trajectory consistency
    \end{itemize}

\end{frame}

% -----------------------------------------------------
\begin{frame}{Final Training Objective}

    \[
        \mathcal{L}
        =
        \mathbb{E}_{i,t}
        \left[
            w_{i,t}
            \cdot
            \mathrm{Huber}
            (f_\theta(x_{i,t}) - y^*_{i,t})
        \right]
    \]

    \vspace{1em}

    \textbf{Key Point}

    \begin{itemize}
        \item Policy training remains standard BC
        \item Robustness handled during preprocessing
    \end{itemize}

\end{frame}

\begin{frame}{Discussion}

    \textbf{Why PACER Works}

    \begin{enumerate}
        \item Progress alignment compares equivalent task stages
        \item Robust statistics resist outliers
        \item Leave-one-out prevents self-validation
        \item Pseudo-label repair suppresses local corruptions
    \end{enumerate}

    \vspace{1em}

    \begin{block}{Interpretability}
        PACER also provides useful diagnostics:
        \begin{itemize}
            \item trust maps
            \item MAD uncertainty regions
            \item phase-aligned consensus trajectories
        \end{itemize}
    \end{block}

\end{frame}

\section{Limitations and Future Work}

% -----------------------------------------------------
\begin{frame}{Limitations}

    \begin{itemize}
        \item Requires meaningful phase estimation
        \item Assumes mostly local corruptions
        \item Sensitive to hyperparameters
        \item Sparse bins reduce robustness
        \item Very few demonstrations weaken LOO consensus
    \end{itemize}

\end{frame}

% -----------------------------------------------------
\begin{frame}{Future Work}

    \begin{itemize}
        \item Vision-based phase estimation
        \item Uncertainty-aware consensus models
        \item Integration with diffusion policies
        \item Online refinement with environment interaction
        \item Validation on real robot systems
    \end{itemize}

\end{frame}
